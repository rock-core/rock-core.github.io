def generate_activity_setup
    if @activity_setup
        @activity_setup.call
    else
        result = <<-EOD
#{activity_type.class_name}* activity_#{name} = new #{activity_type.class_name}(
#{rtt_scheduler},
#{rtt_priority},
task_#{name}->engine(),
"#{name}");
        EOD
    end
end

def rtt_priority
    case @priority
    when :highest
        'RTT::os::HighestPriority'
    when :lowest
        'RTT::os::LowestPriority'
    when Integer
        @priority
    end
end
