def in_context?(*args)
    args = args.to_set
    context.any? do |ctx|
        (args & ctx).size == args.size
    end
end

def remove_context(*args)
    args = args.to_set
    @context = context.dup
    context.delete_if do |ctx|
        (args & ctx).size == args.size
    end
    self
end
