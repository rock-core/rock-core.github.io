def corba_name
    if inlines_code?
        normalize_cxxname(basename)
    elsif contains_opaques?
        "#{corba_namespace}::#{normalize_cxxname(basename.gsub(/[^\w]/, '_'))}_m"
    else
        "#{corba_namespace}::#{normalize_cxxname(basename.gsub(/[^\w]/, '_'))}"
    end
end
